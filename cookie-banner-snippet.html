/* ── COOKIE CONSENT ── */
(function initCookieBanner() {
  var banner = document.getElementById('cookieBanner');
  if (!banner) return;
  var consent = null;
  try { consent = localStorage.getItem('dziel-cookie-consent'); } catch(e) {}
  // Show banner if no decision has been made yet
  if (!consent) {
    setTimeout(function() { banner.classList.add('cb-visible'); }, 1200);
  }
  // If analytics were previously accepted, load GA
  if (consent === 'accepted') { loadAnalytics(); }
})();

function cookieAccept() {
  try { localStorage.setItem('dziel-cookie-consent', 'accepted'); } catch(e) {}
  document.getElementById('cookieBanner').classList.remove('cb-visible');
  loadAnalytics();
}
function cookieDecline() {
  try { localStorage.setItem('dziel-cookie-consent', 'declined'); } catch(e) {}
  document.getElementById('cookieBanner').classList.remove('cb-visible');
}
function loadAnalytics() {
  /* ── HOW TO ADD GOOGLE ANALYTICS ──────────────────────────────
     1. Go to analytics.google.com → create a property for dzielcreative.com
     2. Copy your Measurement ID (looks like: G-XXXXXXXXXX)
     3. Replace G-XXXXXXXXXX below with your real ID
     4. Analytics will ONLY load after user accepts cookies ✓
  ────────────────────────────────────────────────────────────── */
  var GA_ID = 'G-XXXXXXXXXX'; // ← replace with your real GA4 ID
  if (GA_ID === 'G-XXXXXXXXXX') return; // skip until ID is set
  if (document.getElementById('ga-script')) return; // don't load twice
  var s = document.createElement('script');
  s.id = 'ga-script';
  s.async = true;
  s.src = 'https://www.googletagmanager.com/gtag/js?id=' + GA_ID;
  document.head.appendChild(s);
  window.dataLayer = window.dataLayer || [];
  function gtag(){dataLayer.push(arguments);}
  gtag('js', new Date());
  gtag('config', GA_ID, { anonymize_ip: true });
}
